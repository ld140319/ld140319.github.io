<?php
class BinaryTree
{
/**

 * 左节点
 * @var BinaryTree
 */
public $left;

/**
 * 右节点
 * @var BinaryTree
 */
public $right;

/**
 * 树的内容
 * @var mixed
 */
public $data;

public function __construct($left, $right, $data)
{
    $this->left = $left;
    $this->right = $right;
    $this->data = $data;
}

/**
 * @return BinaryTree
 */
public function getLeft()
{
    return $this->left;
}

/**
 * @param BinaryTree $left
 * @return BinaryTree
 */
public function setLeft($left)
{
    $this->left = $left;
    return $this;
}

/**
 * @return BinaryTree
 */
public function getRight()
{
    return $this->right;
}

/**
 * @param BinaryTree $right
 */
public function setRight($right)
{
    $this->right = $right;
}

/**
 * @return mixed
 */
public function getData()
{
    return $this->data;
}

/**
 * @param mixed $data
 */
public function setData($data)
{
    $this->data = $data;
}

/**
 * 插入左节点
 * 如果当前的结点没有左孩子结点，我们就创建一个新结点，然后将其设置为当前结点的左结点。
 * 如果已经有了左结点，我们就创建一个新结点，并将其放在当前左结点的位置。然后再将原左结点值为新左结点的左结点。
 * @param  BinaryTree $node
 * @param mixed $value
 */
public static function insertLeft(BinaryTree $node, $value)
{
    if ($node != null) {
        if ($node->left == null) {
            $node->setLeft($value instanceof BinaryTree ? $value : new BinaryTree(null, null, $value));
        } else {
            $newNode = $value instanceof BinaryTree ? $value : new BinaryTree(null, null, $value);
            $newNode->left = $node->left;
            $node->left = $newNode;
        }
    }
    return $node;
}

/**
 * 插入右节点
 * 如果当前的结点没有右孩子结点，我们就创建一个新结点，然后将其设置为当前结点的右结点。
 * 如果已经有了右结点，我们就创建一个新结点，并将其放在当前右结点的位置。然后再将原左结点值为新右结点的右结点。
 * @param  BinaryTree $node
 * @param mixed $value
 */
public static function insertRight(BinaryTree $node, $value)
{
    if ($node != null) {
        if ($node->right == null) {
            $node->setright($value instanceof BinaryTree ? $value : new BinaryTree(null, null, $value));
        } else {
            $newNode = $value instanceof BinaryTree ? $value : new BinaryTree(null, null, $value);
            $newNode->right = $node->right;
            $node->right = $newNode;
        }
    }
    return $node;
}

/**
 * 先序遍历
 * @param BinaryTree $node
 */
public static function preOrder($node)
{
    $stack = array();
    $data = array();
    while ($node != null || $stack) {
        while ($node != null) {
            $data[] = $node->getData();
            array_push($stack, $node);
            $node = $node->left;
        }
        if ($stack) {
            $node = array_pop($stack);
            $node = $node->right;
        }
    }
    return $data;
}

/**
 * 中序遍历
 * @param BinaryTree $node
 */
public static function inOrder($node)
{
    $stack = array();
    $data = array();
    while ($node != null || $stack) {
        while ($node != null) {
            array_push($stack, $node);
            $node = $node->left;
        }
        if ($stack) {
            $node = array_pop($stack);
            $data[] = $node->getData();
            $node = $node->right;
        }
    }
    return $data;
}

/**
 * 后序遍历
 * @param BinaryTree $node
 */
public static function postOrder($node)
{
    $stack = array();
    $data = array();
    while ($node != null || $stack) {
        while ($node != null) {
            array_push($stack, $node);
            $node = $node->left;
        }
        $flag = true;
        $preNode = null;  // 前驱节点
        while ($stack && $flag == true) {
            // peek 不改变栈的值(不删除栈顶的值)，pop会把栈顶的值删除
            $node = $stack[count($stack)-1];
            if ($node->right == $preNode) { // 之前访问的为空节点或是栈顶节点的右子节点
                $node = array_pop($stack);
                $data[] = $node->getData();
                if (! $stack) {
                    return $data;
                } else {
                    $preNode = $node;
                }
            } else {
                $node = $node->right;
                $flag = false;
            }
        }
    }
    return $data;
}

}
